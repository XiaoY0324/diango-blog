# 安装virtualenv
$ pip install virtualenv  

# 创建虚拟环境
$ virtualenv -p /usr/local/Cellar/python/3.7.0/bin/python3.7 ENV3.7  

# 激活虚拟环境
$ source /Users/ys/python-virtualenv/ENV3.7/bin/activate  

# 查看当前环境下的安装包
$ pip list  
pip (1.5.6)
setuptools (3.6)

# 安装django
$ sudo pipe3 install django
